<script>
  const total = 15;

  function getCount() {
    return document.querySelectorAll('.item.checked').length;
  }

  function updateProgress() {
    const count = getCount();
    const pct = Math.round((count / total) * 100);
    document.getElementById('fill').style.width = pct + '%';
    document.getElementById('label').textContent = `${count} of ${total} checked`;
    document.getElementById('msg').classList.toggle('show', count === total);
  }

  function toggle(el) {
    el.classList.toggle('checked');
    el.querySelector('.checkbox').textContent = el.classList.contains('checked') ? '✓' : '';
    updateProgress();
  }

  function resetAll() {
    document.querySelectorAll('.item').forEach(el => {
      el.classList.remove('checked');
      el.querySelector('.checkbox').textContent = '';
    });
    updateProgress();
  }

  function checkAll() {
    document.querySelectorAll('.item').forEach(el => {
      el.classList.add('checked');
      el.querySelector('.checkbox').textContent = '✓';
    });
    updateProgress();
  }
</script>
